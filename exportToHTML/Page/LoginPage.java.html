17

24           public void login(String mail, String password){
25               driver.findElement(Email).sendKeys(mail);
26               driver.findElement(Password_input).sendKeys(password);
27               driver.findElement(Login_button).click();
28           }
29   
30

33           }

38               Assert.assertEquals(driver.getCurrentUrl(),"https://automationexercise.com/login");
39               Assert.assertTrue(driver.findElement(message).isDisplayed());
40           }
41   
42           public void AssertForEmptyEmailForLogin() {
43               JavascriptExecutor js = (JavascriptExecutor) driver;
44               boolean submitted = (boolean) js.executeScript("return window.formSubmitted;");
45               Assert.assertFalse(submitted);
46           }
47   
48           public void Assertforvalidlogout(){
49               Assert.assertTrue(driver.findElement(Email).isDisplayed());

52           }
